pragma solidity ^0.8.0;

// This contract will "clone" your existing AxiomAnchor contract
import "@openzeppelin/contracts/proxy/Clones.sol";

contract AxiomFactory {
    address public immutable anchorImplementation;
    event RegistryDeployed(address indexed newRegistry, string extension);

    constructor(address _anchorImplementation) {
        anchorImplementation = _anchorImplementation;
    }

    function deployRegistry(string memory extension) external returns (address) {
        // Creates a gas-efficient clone of your AxiomAnchor
        address clone = Clones.clone(anchorImplementation);
        
        // Emit an event so your server knows it happened
        emit RegistryDeployed(clone, extension);
        
        return clone;
    }
}
